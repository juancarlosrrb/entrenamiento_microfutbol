import requests

# Tu token de API de CircleCI
api_token = "TU_TOKEN_DE_API"

# URL base de la API de CircleCI
base_url = "https://circleci.com/api/v2/"

# Realiza una solicitud GET para obtener información sobre tus proyectos
response = requests.get(base_url + "projects", headers={"Circle-Token": api_token})

# Verifica si la solicitud fue exitosa
if response.status_code == 200:
    data = response.json()
    # data ahora contiene la información de tus proyectos
    print(data)
else:
    print("Error al hacer la solicitud a la API de CircleCI:", response.status_code, response.text)
